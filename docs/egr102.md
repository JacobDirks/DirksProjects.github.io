void loop() {
  // put your main code here, to run repeatedly:
  int sensorValue = analogRead(A1); // gets value from IR sensors
  float voltage = sensorValue * (5.0 / 1023.0); // measures the voltage
 // Serial.println(voltage);
if((goodHigh <= voltage) &&( goodLow >= voltage)){Serial.println(voltage);}
else {
  if(goodLow <= voltage){ // alters position as long as this is true
    
    if((pos >= 0)&&(pos<180)){
      boomHeight.write(pos); // starts raising
      delay(30); // gives it a second to move
      pos=pos+1;
    }

  int sensorValue = analogRead(A1); // gets value from IR sensors
  float voltage = sensorValue * (5.0 / 1023.0); // measures the voltage
  Serial.println(voltage);
  Serial.print("loop 1: ");
  }
  if(voltage <= goodHigh) { // alters position as long as it is taller based on voltage return

    if((pos <= 180)&&(pos > 0)){
      boomHeight.write(pos); // starts lowering

      pos = pos-1;
    }
    int sensorValue = analogRead(A1); // gets value from IR sensors
  float voltage = sensorValue * (5.0 / 1023.0); // measures the voltage
